310     /**
311      * This method does parse XML data and push it into the listModel2 JList.
312      */
313     private void feed_xml(String datat, String urli) {
314         //UICruft.info_box("feed_xml: " + datat, "Error");
315         List<ngXMLElement> vals = null;
316         try {
317             vals = ngXML.do_parse(datat);
318         } catch (Exception err) {
319             UICruft.info_box("XML data malformed: " + err.getMessage(), "Error");
320             return;
321         }
322         if (vals == null) {
323             UICruft.info_box("XML data malformed", "Error");
324             return;
325         }
326         int len = vals.size();
327         //UICruft.info_box("len: " + len, "tarkka");
328         String t2[] = new String[len];
329 
330         linklist.setModel(listModel2);
331 
332         /// XXX bgran
333         for (int i = 0; i < len; i++) {
334             ngXMLElement t = vals.get(i);
335             String key = t.getKey();
336             String val = t.getVal();
337             t2[i] = key + " -> " + val + "\n";
338 
339             listModel2.addElement(key + " -> " + val + "\n");
340 
341         }
342         //linklist = new JList(t2);
343         //fileViewer.setText(panedata);
344         //searchResults.removeAll();
345         searchResults.setModel(listModel);
346         //linklist.setModel(listModel2);
347     }

349     @Override
350     public void actionPerformed(ActionEvent e) {
351         //System.out.println(e.getActionCommand());
352         switch (e.getActionCommand()) {
353             case "GetRSS":
354                 String search_word = searchField.getText();
355                 file_locker.add_stream(search_word);
356 
357                 refresh_RSS_list();
358                 //listModel = new DefaultListModel();
359                 break;
360             case "RefreshFeeds":
361                 //
362                 //
363                 UICruft.info_box("RefreshFeeds", "Not implemented!");
364                 break;
365             case "Exit":
366                 System.exit(0);
367                 break;
368             case "ClearList":
369                 //UICruft.info_box("Clears the feed list!", "Not implemented!");
370                 file_locker.clean_streams();
371                 delete_RSS_list();
372 
373                 break;
374             case "ClearItem":
375                 int index = searchResults.getSelectedIndex();
376                 if (index == -1) {
377                     break;
378                 }
379                 //System.out.println("Index selected: " + index);
380                 String s = (String) searchResults.getSelectedValue();
381                 //System.out.println("Value selected: " + s);
382                 //UICruft.info_box("index: " + index, "DEBUG");
383                 delete_RSS_item(index);
384                 break;
385             case "OpenHelp":
386                 about_window();
387                 break;
388         }
389         this.repaint();
390     }
