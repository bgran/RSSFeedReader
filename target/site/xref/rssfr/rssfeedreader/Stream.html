40      /**
41       * High level method to fetch data from an URL.
42       */
43      public void setup_content() {
44          String data = get_HTTP_content();
45          List<ngXMLElement> vals = null;
46  
47          try {
48              vals = ngXML.do_parse(data);
49          } catch (Exception err) {
50              UICruft.info_box("XML p broke: " + err.getMessage(),
51                      "XML error");
52              return;
53          }
54  
55      }
56  
57      /**
58       * Sets up a HTTP connection and fetches the raw data.
59       *
60       * @return The URL content as String
61       */
62      public String get_HTTP_content() {
63          this.netw.setUrlStr(rss_url);
64          try {
65              this.netw.init_url_connection();
66              this.netw.init_io_connection();
67              return this.netw.network_get_content();
68          } catch (Exception e) {
69              UICruft.info_box("Error fetching content: " + e.getMessage(), "ERROR");
70              return (null);
71          }
72      }
