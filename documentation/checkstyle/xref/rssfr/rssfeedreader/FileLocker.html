3   import java.io.IOException;
4   import java.util.ArrayList;

7   /**
8    * This class is a holder of some stream related data.
9    */
10  public class FileLocker {
11  
12      final int max_number_of_stream = 100;
13  
14      public ArrayList<String> my_streams2;
15  
16      /**
17       * The constructor tries to load from a "config.ini" file the previously
18       * used RSS feeds.
19       * @param init_ To load feeds from config.ini or not.
20       */
21      public FileLocker(boolean init_) {
22          my_streams2 = new ArrayList<String>();
23          if (init_) {
24              load_from_file();
25          }
26      }
27  
28      /**
29       * add_stream adds a string to be shown in the upper pane.
30       *
31       * @param url The locator to be added.
32       */
33      public void add_stream(String url) {
34  
35          my_streams2.add(url);
36          load_to_file();
37          //UICruft.info_box("add_stream: " + url, "Fii");
38          //UICruft.info_box("kala: " + my_streams2.get(0), "HUOH");
39          //int i;
40          //for (i = 0; my_streams[i] != null; i++);
41          //my_streams[i] = url;
42      }
43  
44      /**
45       * Clears all streams from being shown in the ui.
46       */
47      public void clean_streams() {
48          my_streams2.clear();
49          //for (int i = 0; i < max_number_of_stream; i++) {
50          //    my_streams[i] = null;
51          //}
52          load_to_file();
53      }
54  
55      /**
56       * Remove a RSS entry from the upper pane, by index.
57       *
58       * @param idx The index of the RSS feed in the upper pane to remove.
59       */
60      public void remove_index(int idx) {
61          my_streams2.remove(idx);
62          load_to_file();
63      }
64  
65      /**
66       * Tries to load from the "config.ini" file the previously used RSS feeds.
67       */
68      private void load_from_file() {
69          String data;
70          try {
71              data = Cruft.read_file("config.ini");
72          } catch (IOException e) {
73              System.out.println("Could not load init file");
74              return;
75          }
76          String[] entries = data.split("\n");
77          for (int i = 0; i < entries.length; i++) {
78              if (entries[i].isEmpty()) {
79                  continue;
80              }
81              add_stream(entries[i]);
82          }
83      }
84  
85      /**
86       * Tries to load to the "config.ini" file the previously used RSS feeds.
87       */
88      private void load_to_file() {
89          try {
90              Cruft.write_file_spec("config.ini", my_streams2);
91          } catch (IOException e) {
92              System.out.println(
93                      "Could not write config.ini:" + e.getMessage());
94          }
95      }
96  }
